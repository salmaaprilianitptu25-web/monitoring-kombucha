#include #include #include #include #include #include #include #include #include #include #define WIFI_SSID "april" #define WIFI_PASSWORD "iyaituaja" #define FIREBASE_HOST "kombucha-iot-bff2b-default-rtdb.asia-southeast1.firebasedatabase.app" #define FIREBASE_AUTH "" #define PIN_MQ135 34 #define PIN_DS18B20 4 #define PIN_RELAY 2 #define SD_CS 5 LiquidCrystal_I2C lcd(0x27, 20, 4); OneWire oneWire(PIN_DS18B20); DallasTemperature sensors(&oneWire); FirebaseData fbdo; FirebaseAuth auth; FirebaseConfig config; MQUnifiedsensor MQ135("ESP-32", 3.3, 12, PIN_MQ135, "MQ-135"); String statusActuator = "OFF"; bool isSystemRunning = true; // Variabel untuk start/stop void setup() { Serial.begin(115200); pinMode(PIN_RELAY, OUTPUT); digitalWrite(PIN_RELAY, LOW); lcd.init(); lcd.backlight(); lcd.setCursor(0,0); lcd.print(" SYSTEM STARTING "); WiFi.begin(WIFI_SSID, WIFI_PASSWORD); while (WiFi.status() != WL_CONNECTED) { delay(500); } config.host = FIREBASE_HOST; config.signer.tokens.legacy_token = FIREBASE_AUTH; Firebase.begin(&config, &auth); Firebase.reconnectWiFi(true); SD.begin(SD_CS); sensors.begin(); MQ135.setRegressionMethod(1); MQ135.setA(110.47); MQ135.setB(-2.862); MQ135.init(); MQ135.setR0(62.04); lcd.clear(); } void loop() { // 1. CEK PERINTAH DARI WEB (START/STOP) if (Firebase.getString(fbdo, "/kombucha_control/command")) { String cmd = fbdo.stringData(); if (cmd == "STOP") isSystemRunning = false; else if (cmd == "START") isSystemRunning = true; } if (!isSystemRunning) { lcd.setCursor(0,0); lcd.print("STATUS: SYSTEM STOP "); lcd.setCursor(0,1); lcd.print("Waiting Start... "); digitalWrite(PIN_RELAY, LOW); delay(2000); return; // Berhenti di sini, tidak baca sensor } // 2. BACA SENSOR MQ135.update(); float co2 = MQ135.readSensor(); if (co2 < 400) co2 = 400; sensors.requestTemperatures(); float t1 = sensors.getTempCByIndex(0); float t2 = sensors.getTempCByIndex(1); // 3. LOGIKA RELAY if (co2 > 1000) { digitalWrite(PIN_RELAY, HIGH); statusActuator = "ON "; } else { digitalWrite(PIN_RELAY, LOW); statusActuator = "OFF"; } // 4. UPDATE LCD lcd.setCursor(0, 0); lcd.print("CO2: "); lcd.print((int)co2); lcd.print(" PPM "); lcd.setCursor(0, 1); lcd.print("T1 : "); lcd.print(t1, 1); lcd.print("C T2: "); lcd.print(t2, 1); lcd.setCursor(0, 2); lcd.print("Actuator: "); lcd.print(statusActuator); lcd.setCursor(0, 3); lcd.print("WiFi: OK SD: LOGGING"); // 5. KIRIM KE FIREBASE & SD Firebase.setInt(fbdo, "/kombucha/co2", (int)co2); Firebase.setFloat(fbdo, "/kombucha/suhu1", t1); Firebase.setFloat(fbdo, "/kombucha/suhu2", t2); Firebase.setString(fbdo, "/kombucha/status_relay", statusActuator); File dataFile = SD.open("/log.csv", FILE_APPEND); if (dataFile) { dataFile.printf("%lu,%d,%.1f,%.1f,%s\n", millis(), (int)co2, t1, t2, statusActuator.c_str()); dataFile.close(); } delay(5000); }
